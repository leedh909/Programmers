SELECT B.LV HOUR, NVL(A.COUNT,0) COUNT
FROM (
        SELECT TO_CHAR(DATETIME,'HH24') HOUR, COUNT(*) COUNT
        FROM ANIMAL_OUTS
        GROUP BY TO_CHAR(DATETIME,'HH24')
        ORDER BY HOUR
    )A, ( 
        SELECT (LEVEL-1) AS LV FROM DUAL 
        CONNECT BY LEVEL<=24
    )B
WHERE B.LV = A.HOUR(+)
ORDER BY B.LV
